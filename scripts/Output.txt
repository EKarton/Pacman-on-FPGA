			next_reset_state = cur_reset_state + 1;
		end
	end
end

always @(posedge clock_50)
begin
	if(reset_n == 1'b1)
	begin
		cur_reset_state <= next_reset_state;
	end
	else
	begin
		cur_reset_state <= 8'd0;
		next_reset_state <= 8'd0;
	end
end
